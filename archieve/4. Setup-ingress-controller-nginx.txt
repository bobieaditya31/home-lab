# Setup Ingress Controller
kubectl apply -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/main/deploy/static/provider/kind/deploy.yaml

# Running Patch
cd /home/bobski/Documents/home-lab/cluster/
./ingress-nginx-patch.sh 

kubectl wait --namespace ingress-nginx \
  --for=condition=ready pod \
  --selector=app.kubernetes.io/component=controller \
  --timeout=120s
  
# Verifikasi
kubectl get pods -n ingress-nginx

# Install menggunakan helm
helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
helm repo update

helm install ingress-nginx ingress-nginx/ingress-nginx \
  -n ingress-nginx \
  --create-namespace \
  -f ingress-nginx-values.yaml

# Cek Port NodePort yang Ter-assign
kubectl get svc -n ingress-nginx
kubectl get pods,svc,ingress -n argocd
kubectl get pod -n ingress-nginx -o wide

# Delete namespace
kubectl delete namespace ingress-nginx


# Retart pod
kubectl rollout restart deploy ingress-nginx-controller -n ingress-nginx

# Install menggunakan file yaml lokal
kubectl apply -f ingress-nginx.yaml
